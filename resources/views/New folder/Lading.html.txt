            });

        // Simple counter animation for stats
        const counters = document.querySelectorAll('.font-stats-number');
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    const target = entry.target;
                    const finalValue = parseInt(target.innerText.replace(',', ''));
                    let count = 0;
                    const duration = 2000;
                    const increment = Math.ceil(finalValue / (duration / 16));
                    
                    const updateCount = () => {
                        count += increment;
                        if (count < finalValue) {
                            target.innerText = count.toLocaleString();
                            requestAnimationFrame(updateCount);
                        } else {
                            target.innerText = finalValue.toLocaleString();
                        }
                    };
                    updateCount();
                    observer.unobserve(target);
                }
            });
        }, { threshold: 0.5 });

        counters.forEach(counter => observer.observe(counter));
